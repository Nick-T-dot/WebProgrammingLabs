int getNthPiDigit(int n)
{
	double ans = 0;
	for (int i = 0; i < n; i++)
	{
		ans += 1 / pow(16, i) * (4.0 / (8*i + 1) - 2.0 / (8*i + 4) - 1.0 / (8*i + 5) - 1.0 / (8*i + 6));
	}
	ans = trunc((pow(10,n - 1) * (ans) - trunc(pow(10,n - 1) * (ans))) * 10);
	return ans;
}
